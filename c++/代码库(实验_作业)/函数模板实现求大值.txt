#include<iostream>
using namespace std;
template <typename t>
t max(const t a,const t b)
{return a>b?a:b;}

int main()
{
	int j,k;
	cout<<"enter two integer:"<<endl;
	cin>>j>>k;
	cout<<"max("<<j<<","<<k<<")="<<max(j,k)<<endl;
	double x,y;
	cout<<"enter two double:"<<endl;
	cin>>x>>y;
	cout<<"max("<<x<<","<<y<<")="<<max(x,y)<<endl;
	char c,h;
	cin>>c>>h;
	cout<<"max("<<x<<","<<y<<")="<<max(x,y)<<endl;
}
